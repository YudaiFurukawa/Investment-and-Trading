class TimeSeriesSplit(n):
def init(self, n_splits=3):
super(TimeSeriesSplit, self).init(n_splits,shuffle=False,random_state=None)

def split(X, y=None, groups=None):
    X, y, groups = indexable(X, y, groups)
    n_samples = _num_samples(X)
    n_splits = self.n_splits
    n_folds = n_splits + 1
    if n_folds > n_samples:
        raise ValueError(
            ("Cannot have number of folds ={0} greater"
             " than the number of samples: {1}.").format(n_folds,n_samples))
    indices = np.arange(n_samples)
    test_size = (n_samples // n_folds)
    test_starts = range(test_size + n_samples % n_folds,
                        n_samples, test_size)
    for test_start in test_starts:
        yield (indices[:test_start],
            indices[test_start:test_start + test_size])
